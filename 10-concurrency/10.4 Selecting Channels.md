func (job *TaxIncludedPriceJob) Process(doneChan chan bool, errorChan chan error) error {
	err := job.LoadData()
	if err != nil {
		errorChan <- err
		return err
	}

	result := make(map[string]string)
	for _, price := range job.InputPrices {
		taxIncludedPrice := price * (1 + job.TaxRate)
		result[fmt.Sprintf("%.2f", price)] = fmt.Sprintf("%.2f", taxIncludedPrice)
	}

	job.TaxIncludedPrices = result
	err = job.IOManager.WriteResult(job)
	if err != nil {
		errorChan <- err
		return err
	}

	doneChan <- true
	return nil
}

We can’t just loop over `errorChan` and `doneChan` separately because each goroutine only sends to only one of them, either an error or a success, and both channel operations are blocking. That means the program ends up waiting for messages that will never be sent, causing a deadlock. Instead, we need a way to listen to both channels at the same time and handle whichever message arrives, without getting stuck waiting unnecessarily. 

We can use the `select` control structure to help here. `select` is similar to `switch` except it waits for different values emitted for a channel, switching behaviour based on the channel that emitted the data.

`main.go`:
```go
package main

import (
	"fmt"

	"github.com/Tyler-Liquornik/golang-fundamentals/10-concurrency/10.3-10.4/filemanager"
	"github.com/Tyler-Liquornik/golang-fundamentals/10-concurrency/10.3-10.4/prices"
)

func main() {
	taxRates := []float64{0, 0.07, 0.1, 0.15}
	doneChans := make([]chan bool, len(taxRates))
	errorChans := make([]chan error, len(taxRates))

	for i, taxRate := range taxRates {
		doneChans[i] = make(chan bool)
		errorChans[i] = make(chan error)

		inputPath := "10-concurrency/10.3-10.5/prices/prices.txt"
		outputPath := fmt.Sprintf("10-concurrency/10.3-10.5/results/result_%.0f.json", taxRate*100)
		fm := filemanager.New(inputPath, outputPath)

		// cmdm := cmdmanager.New() can get input from cmdm now instead of from files

		priceJob := prices.NewTaxIncludedPriceJob(fm /*cmdm*/, taxRate)
		go priceJob.Process(doneChans[i], errorChans[i])
	}

	for i := range taxRates {
		select {
		case err := <-errorChans[i]:
			if err != nil {
				fmt.Println("Error processing prices:", err)
			}
		case <-doneChans[i]:
			fmt.Println("Prices processed successfully!")
		}
	}
}
```
