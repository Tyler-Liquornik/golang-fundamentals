Lets add some multi-threaded features to the large project from the previous section.

Let's simulate a very slow file write first to create a simulated use case for concurrent jobs.

`filemanager.go` (snippet):
```go
func (fm FileManager) WriteResult(data any) error {
	file, err := os.Create(fm.OutputFilePath)
	if err != nil {
		fmt.Println("Error creating file:", err)
		return errors.New("failed to create file")
	}
	
	time.Sleep(3 * time.Second)

	encoder := json.NewEncoder(file)
	err = encoder.Encode(data)
	if err != nil {
		fmt.Println("Error writing JSON data:", err)
		file.Close()
		return errors.New("failed to write JSON data")
	}

	file.Close()
	return nil
}
```

`main.go`:
```go
package main

import (
	"fmt"

	"github.com/Tyler-Liquornik/golang-fundamentals/10-concurrency/10.3/filemanager"
	"github.com/Tyler-Liquornik/golang-fundamentals/10-concurrency/10.3/prices"
)

func main() {
	taxRates := []float64{0, 0.07, 0.1, 0.15}
	for _, taxRate := range taxRates {
		inputPath := "10-concurrency/10.3-10.5/prices/prices.txt"
		outputPath := fmt.Sprintf("10-concurrency/10.3-10.5/results/result_%.0f.json", taxRate*100)
		fm := filemanager.New(inputPath, outputPath)

		// cmdm := cmdmanager.New() can get input from cmdm now instead of from files

		priceJob := prices.NewTaxIncludedPriceJob(fm /*cmdm*/, taxRate)
		err := priceJob.Process()
		if err != nil {
			fmt.Println("Error processing prices:", err)
			continue
		}
	}
}
```

This will take 12 seconds to run `main.go`, 3 seconds per tax rate.

Now let's see if we can run the price jobs concurrently and then the whole thing should take only 3 seconds with all 4 jobs running at the same time. This is a huge time savings. If we write `go priceJob.Process()`, we're going to need to make `Process` not return an error anymore, since goroutines can't return data, they must transmit it through channels.

`prices.go` (snippet):
```go
func (job *TaxIncludedPriceJob) Process(doneChan chan bool) error {
	err := job.LoadData()
	if err != nil {
		return err
	}

	result := make(map[string]string)
	for _, price := range job.InputPrices {
		taxIncludedPrice := price * (1 + job.TaxRate)
		result[fmt.Sprintf("%.2f", price)] = fmt.Sprintf("%.2f", taxIncludedPrice)
	}

	job.TaxIncludedPrices = result
	err = job.IOManager.WriteResult(job)
	if err != nil {
		return err
	}
	
	doneChan <- true
	return nil
}
```

> Notice how we have a traditional `return` statement, and also end off emitting a value to our channel. This is so that `Process` can be used either as a normal function, or as a goroutine, just remember if used as a goroutine the nothing is actually returned.

`main.go`:
```go
package main

import (
	"fmt"

	"github.com/Tyler-Liquornik/golang-fundamentals/10-concurrency/10.3/filemanager"
	"github.com/Tyler-Liquornik/golang-fundamentals/10-concurrency/10.3/prices"
)

func main() {
	taxRates := []float64{0, 0.07, 0.1, 0.15}
	doneChans := make([]chan bool, len(taxRates))

	for i, taxRate := range taxRates {
		doneChans[i] = make(chan bool)

		inputPath := "10-concurrency/10.3-10.5/prices/prices.txt"
		outputPath := fmt.Sprintf("10-concurrency/10.3-10.5/results/result_%.0f.json", taxRate*100)
		fm := filemanager.New(inputPath, outputPath)

		// cmdm := cmdmanager.New() can get input from cmdm now instead of from files

		priceJob := prices.NewTaxIncludedPriceJob(fm /*cmdm*/, taxRate)
		go priceJob.Process(doneChans[i])

		//if err != nil {
		//	fmt.Println("Error processing prices:", err)
		//	continue
		//}
	}

	for _, doneChan := range doneChans {
		<-doneChan
	}
}
```

Result: takes 3 seconds instead of 12 seconds.
